</> Code Examples

Start building in minutes

Copy-paste examples to get up and running fast. Every example works out of the box with your API key.

translate_text.py 
import deepl

translator = deepl.Translator("YOUR_API_KEY")

result = translator.translate_text(
    "The quick brown fox jumps over the lazy dog.",
    target_lang="DE",
    formality="more"
)

print(result.text)
# → "Der schnelle braune Fuchs springt über den faulen Hund."
translate_text.js 
import * as deepl from 'deepl-node';

const translator = new deepl.Translator('YOUR_API_KEY');

const result = await translator.translateText(
    'The quick brown fox jumps over the lazy dog.',
    null,
    'de',
    { formality: 'more' }
);

console.log(result.text);
// → "Der schnelle braune Fuchs springt über den faulen Hund."
translate_text.sh 
curl https://api.deepl.com/v2/translate \
  -H "Authorization: DeepL-Auth-Key YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "text": ["The quick brown fox jumps over the lazy dog."],
    "target_lang": "DE",
    "formality": "more"
  }'

translate_doc.sh 
# Step 1: Upload document
curl https://api.deepl.com/v2/document \
  -H "Authorization: DeepL-Auth-Key YOUR_API_KEY" \
  -F "file=@report.pdf" \
  -F "target_lang=DE"

# Step 2: Poll status (use document_id from response)
curl https://api.deepl.com/v2/document/{id} \
  -H "Authorization: DeepL-Auth-Key YOUR_API_KEY" \
  -d '{"document_key": "..."}'

# Step 3: Download translated document
curl https://api.deepl.com/v2/document/{id}/result \
  -H "Authorization: DeepL-Auth-Key YOUR_API_KEY" \
  -d '{"document_key": "..."}' -o report_de.pdf
batch_voice.py 
import deepl

client = deepl.VoiceClient("YOUR_API_KEY")

# Transcribe + translate an audio file
result = client.translate_audio(
    file_path="meeting_en.mp3",
    target_lang="DE",
    output_format="srt"
)

print(result.transcript)
result.save_subtitles("meeting_de.srt")
batch_voice.js 
import { VoiceClient } from 'deepl-node';

const client = new VoiceClient('YOUR_API_KEY');

// Transcribe + translate an audio file
const result = await client.translateAudio('meeting_en.mp3', {
    targetLang: 'de',
    outputFormat: 'srt'
});

console.log(result.transcript);
await result.saveSubtitles('meeting_de.srt');

  -F "file=@meeting_en.mp3" \
  -F "target_lang=DE" \
  -F "output_format=srt"

Featured: GitHub Actions Popular

.github/workflows/translate.yml 
name: Translate i18n files
on:
  push:
    paths: ['locales/en/**']

jobs:
  translate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: deepl-com/translate-action@v1
        with:
          api-key: ${{ secrets.DEEPL_API_KEY }}
          source-path: locales/en
          target-langs: de,fr,ja,ko,zh
      - uses: peter-evans/create-pull-request@v6
        with:
          title: "chore: update translations"
